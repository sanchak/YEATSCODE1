ebegin{table}
\caption {\bf Definitions of Specificity, Sensitivity and False Positive Rate}
\begin{tabular}{ p{2cm}   p{7cm}     } 
\hline
	TP & True Postives \\
\hline
	FP & False Postives \\
\hline
	TN & True Negatives \\
\hline
	FN & False Negatives \\
\hline
               Sensitivity &  $\frac{TP}{TP + FN} $ \\
\hline
Specificity &  $\frac{TN}{FP + TN}$    \\
\hline
False Positive Rate &  $1-Specificity$   \\
\hline
\end{tabular} \label{table:definitions}
\end{table}
